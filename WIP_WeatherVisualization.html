weather_data <- weather_data %>% 
     mutate(DATE = ymd(DATE)) %>% 
     mutate(date = lubridate::ymd(DATE), year = lubridate::year(DATE), month = lubridate::month(DATE)) %>% 
     select(-DATE) %>%
     mutate(TAVG = (TMAX + TMIN)/2) %>% 
     select(date, year, month, TAVG:TMIN, everything())
